
port input  n        [7:0]  # N'th number of the sequence to calculate
port input  valid           # Is the input N valid?
port output done            # Have we finished computing the result?
port output result   [63:0] # The result of calculating the n'th number.

state   counter      [7:0]
state   num_a        [63:0]
state   num_b        [63:0]
state   tmp_a        [63:0]

block main
    
    set         done    0
    set         num_a   0
    set         num_b   1
    set         counter 1

    ifeqz       valid   main
    goto        compute

block compute

    copy        counter n
    goto        compute_loop

block compute_loop
    copy        tmp_a   num_a
    add_vars    num_a   num_a   num_b
    copy        num_b   tmp_a
    dec         counter 1
    goto        continue

block continue
    ifnez       counter compute_loop
    goto        finished

block finished
    copy        result  num_a
    set         done    1
    ifeqz       valid   finished
    goto        main
